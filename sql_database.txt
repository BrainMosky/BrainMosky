CREATE TABLE cards (
    id INTEGER,
    card_name TEXT,
    strength INTEGER,
    life INTEGER,
    card_type INTEGER,
    PRIMARY KEY(id)
);
CREATE TABLE decks (
    deck_id INTEGER,
    card_id INTEGER,
    FOREIGN KEY(card_id) REFERENCES cards(id)
    FOREIGN KEY(deck_id) REFERENCES deck_names(deck_id)
);
CREATE TABLE deck_names (
    deck_id INTEGER,
    name TEXT,
    PRIMARY KEY(deck_id)
);
